        function loadUserData() {
            const localData = localStorage.getItem('JAC_PRO_USER_SCORES_V5');
            if (localData) {
                userScoresData = JSON.parse(localData);
            } else {
                userScoresData = {
                    "gakka-1": 45,
                    "gakka-2": 70
                };
            }
            recalculateCategoryDiagnostics();
            updateGlobalStatsBar();
        }

        function saveUserData() {
            localStorage.setItem('JAC_PRO_USER_SCORES_V5', JSON.stringify(userScoresData));
            recalculateCategoryDiagnostics();
            updateGlobalStatsBar();
        }

        function recalculateCategoryDiagnostics() {
            const categories = ["safety", "tools", "scaffolding", "concrete"];
            const keys = Object.keys(userScoresData);
            if (keys.length === 0) {
                categories.forEach(c => {
                    document.getElementById(`diag-pct-${c === "scaffolding" ? "scaff" : c}`).innerText = "0%";
                    document.getElementById(`diag-bar-${c === "scaffolding" ? "scaff" : c}`).style.width = "0%";
                });
                return;
            }

            let mockWeights = { safety: 0, tools: 0, scaffolding: 0, concrete: 0 };
            keys.forEach(k => {
                const s = userScoresData[k];
                mockWeights.safety += s >= 65 ? 85 : 45;
                mockWeights.tools += s >= 65 ? 75 : 40;
                mockWeights.scaffolding += s >= 65 ? 90 : 35;
                mockWeights.concrete += s >= 65 ? 80 : 30;
            });

            categories.forEach(c => {
                const rawVal = Math.min(Math.round(mockWeights[c] / keys.length), 100);
                const tag = c === "scaffolding" ? "scaff" : c;
                document.getElementById(`diag-pct-${tag}`).innerText = `${rawVal}%`;
                document.getElementById(`diag-bar-${tag}`).style.width = `${rawVal}%`;
            });

            let sorted = categories.map(c => ({
                cat: c,
                val: Math.min(Math.round(mockWeights[c] / keys.length), 100)
            })).sort((a,b) => a.val - b.val);

            const catMap = {
                safety: "An Toàn Lao Động 🛡️",
                tools: "Sử Dụng Thiết Bị 🛠️",
                scaffolding: "Giàn Giáo Cao Trình 🪜",
                concrete: "Bê Tông Cốt Thép 🧱"
            };
            const weakestText = sorted[0] ? `${catMap[sorted[0].cat]} (${sorted[0].val}%)` : "Tính toán...";
            document.getElementById('ai-quiz-weakest-tag').innerText = "Trọng tâm: " + weakestText;
            document.getElementById('ai-quiz-weakest-tag').setAttribute('data-weakest', sorted[0] ? sorted[0].cat : "safety");
        }

        function updateGlobalStatsBar() {
            const keys = Object.keys(userScoresData);
            const totalCount = keys.length;
            document.getElementById('stat-completed-desc').innerText = `${totalCount} / 100`;

            if (totalCount > 0) {
                let sum = 0;
                let passedCount = 0;
                keys.forEach(k => {
                    const sc = userScoresData[k];
                    sum += sc;
                    if (sc >= 65) passedCount++;
                });
                const avg = Math.round(sum / totalCount);
                document.getElementById('stat-avg-desc').innerText = `${avg}%`;

                let rank = "Kỹ sư Tập sự";
                if (passedCount >= 40) rank = "Huyền Thoại JAC 👑";
                else if (passedCount >= 20) rank = "Đốc Công Công Trường 🏗️";
                else if (passedCount >= 5) rank = "Thợ Lành Nghề 👷";
                document.getElementById('stat-rank-desc').innerText = rank;
            } else {
                document.getElementById('stat-avg-desc').innerText = "0%";
                document.getElementById('stat-rank-desc').innerText = "Kỹ sư Tập sự";
            }
        }

        function renderHubSets() {
            const container = document.getElementById('sets-grid-container');
            container.innerHTML = '';
            const filterVal = document.getElementById('set-filter-select').value;

            for (let i = 1; i <= 50; i++) {
                const scoreKey = `${activeHubType}-${i}`;
                const bestScore = userScoresData[scoreKey];
                const hasAttempted = bestScore !== undefined;
                const isPassed = hasAttempted && bestScore >= 65;

                if (filterVal === 'unattempted' && hasAttempted) continue;
                if (filterVal === 'passed' && !isPassed) continue;
                if (filterVal === 'failed' && (!hasAttempted || isPassed)) continue;

                const card = document.createElement('div');
                card.className = `p-4.5 rounded-2xl border transition duration-200 flex flex-col justify-between h-40 ${
                    isPassed 
                        ? 'bg-emerald-50/40 border-emerald-100 hover:border-emerald-300 shadow-sm' 
                        : hasAttempted 
                            ? 'bg-rose-50/40 border-rose-100 hover:border-rose-300' 
                            : 'bg-slate-50/60 border-slate-200 hover:border-slate-300'
                }`;

                let badge = `<span class="px-2 py-0.5 rounded text-[9px] font-bold bg-slate-200 text-slate-600">Chưa làm</span>`;
                if (hasAttempted) {
                    badge = isPassed 
                        ? `<span class="px-2 py-0.5 rounded text-[9px] font-bold bg-emerald-100 text-emerald-800"><i class="fa-solid fa-circle-check"></i> Đạt (${bestScore}%)</span>`
                        : `<span class="px-2 py-0.5 rounded text-[9px] font-bold bg-rose-100 text-rose-800"><i class="fa-solid fa-circle-xmark"></i> Hỏng (${bestScore}%)</span>`;
                }

                card.innerHTML = `
                    <div>
                        <div class="flex justify-between items-center mb-2">
                            <span class="text-xs font-black text-slate-400">SET ${String(i).padStart(2, '0')}</span>
                            ${badge}
                        </div>
                        <h4 class="font-extrabold text-xs text-slate-800 leading-tight">Bộ Đề Sát Hạch Số ${i}</h4>
                        <p class="text-[10px] text-slate-400 mt-1">${activeHubType === 'gakka' ? '40 câu • 60 phút' : '25 câu • 40 phút'}</p>
                    </div>
                    <button onclick="startSpecificExam('${activeHubType}', ${i})" class="w-full py-2 bg-slate-900 hover:bg-slate-800 text-white rounded-xl text-xs font-black transition-all">
                        Khởi động đề
                    </button>
                `;
                container.appendChild(card);
            }
        }

        function filterSets() {
            renderHubSets();
        }

        async function generateAiQuiz() {
            const apiKey = localStorage.getItem('JAC_PREP_API_KEY') || "";
            if (!apiKey) {
                document.getElementById('ai-quiz-warning-box').classList.remove('hidden');
                showCustomAlert("Vui lòng cấu hình API Key để kích hoạt Trình Sinh Đề Tự Động bằng AI.");
                return;
            } else {
                document.getElementById('ai-quiz-warning-box').classList.add('hidden');
            }

            document.getElementById('ai-quiz-setup').classList.add('hidden');
            document.getElementById('ai-quiz-loader').classList.remove('hidden');

            const weakestCategory = document.getElementById('ai-quiz-weakest-tag').getAttribute('data-weakest') || "safety";
            const categoryPromptText = {
                safety: "An toàn vệ sinh lao động và sức khỏe công trường",
                tools: "Sử dụng thiết bị, dụng cụ cầm tay an toàn",
                scaffolding: "Giàn giáo, giá đỡ, thang dốc leo cao",
                concrete: "Ván khuôn, bê tông tươi và gia cố cốt thép dầm sàn"
            }[weakestCategory];

            const modelUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${apiKey}`;
            const prompt = `Hãy đóng vai giảng viên cao cấp JAC của Nhật Bản. Tạo 5 câu hỏi trắc nghiệm tiếng Nhật chuyên ngành xây dựng tập trung 100% vào chuyên đề: "${categoryPromptText}".
            Yêu cầu:
            1. Các câu hỏi viết bằng tiếng Nhật chuẩn, có phiên âm Hiragana trên đầu chữ Kanji sử dụng thẻ <ruby> (Ví dụ: <ruby>整理<rt>せいり</rt></ruby>).
            2. Có bản dịch nghĩa câu hỏi sang tiếng Việt.
            3. Cung cấp 3 phương án lựa chọn, ký hiệu bắt đầu bằng A, B, C.
            4. Chỉ ra ký hiệu đáp án đúng (A, B, hoặc C).
            5. Cung cấp giải thích tiếng Nhật và tiếng Việt chi tiết.`;

            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: "Bạn là trợ lý JAC chuyên tạo câu hỏi trắc nghiệm tiếng Nhật có ruby furigana. Chỉ trả về dữ liệu định dạng JSON hợp lệ." }] },
                generationConfig: {
                    responseMimeType: "application/json",
                    responseSchema: {
                        type: "OBJECT",
                        properties: {
                            questions: {
                                type: "ARRAY",
                                items: {
                                    type: "OBJECT",
                                    properties: {
                                        q: { type: "STRING", description: "Câu hỏi tiếng Nhật bọc thẻ ruby furigana hoàn chỉnh" },
                                        viet: { type: "STRING", description: "Bản dịch nghĩa câu hỏi sang tiếng Việt" },
                                        choices: {
                                            type: "ARRAY",
                                            items: { type: "STRING" },
                                            description: "Danh sách 3 phương án lựa chọn dạng A., B., C."
                                        },
                                        ans: { type: "STRING", description: "Đáp án đúng chỉ chứa chữ cái A, B hoặc C" },
                                        exp: { type: "STRING", description: "Chữa giải thích chi tiết bằng tiếng Nhật" },
                                        expViet: { type: "STRING", description: "Chữa giải thích chi tiết bằng tiếng Việt" }
                                    },
                                    required: ["q", "viet", "choices", "ans", "exp", "expViet"]
                                }
                            }
                        },
                        required: ["questions"]
                    }
                }
            };

            let success = false;
            let questionsData = [];
            let delay = 1000;

            for (let attempt = 1; attempt <= 4; attempt++) {
                try {
                    const response = await fetch(modelUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (response.ok) {
                        const data = await response.json();
                        const rawText = data.candidates?.[0]?.content?.parts?.[0]?.text;
                        if (rawText) {
                            const parsed = JSON.parse(rawText);
                            questionsData = parsed.questions || [];
                            if (questionsData.length > 0) {
                                success = true;
                                break;
                            }
                        }
                    }
                } catch (err) {
                    // silent backoff
                }
                if (attempt < 4) {
                    await new Promise(resolve => setTimeout(resolve, delay));
                    delay *= 2;
                }
            }

            document.getElementById('ai-quiz-loader').classList.add('hidden');

            if (success && questionsData.length > 0) {
                activeAiQuizQuestions = questionsData;
                activeAiQuizIdx = 0;
                activeAiQuizAnswers = {};

            }
        }

        function convertPcmToWav(base64Data, sampleRate) {
            const rawBuffer = base64ToArrayBuffer(base64Data);
            const pcm16 = new Int16Array(rawBuffer);
            
            const buffer = new ArrayBuffer(44 + pcm16.length * 2);
            const view = new DataView(buffer);
            
            writeString(view, 0, 'RIFF');
            view.setUint32(4, 36 + pcm16.length * 2, true);
            writeString(view, 8, 'WAVE');
            writeString(view, 12, 'fmt ');
            view.setUint32(16, 16, true);
            view.setUint16(20, 1, true);
            view.setUint16(22, 1, true);
            view.setUint32(24, sampleRate, true);
            view.setUint32(28, sampleRate * 2, true);
            view.setUint16(32, 2, true);
            view.setUint16(34, 16, true);
            writeString(view, 36, 'data');
            view.setUint32(40, pcm16.length * 2, true);
            
            for (let i = 0; i < pcm16.length; i++) {
                view.setInt16(44 + i * 2, pcm16[i], true);
            }
            
            return new Blob([view], { type: 'audio/wav' });
        }

        function base64ToArrayBuffer(base64) {
            const binaryString = atob(base64);
            const len = binaryString.length;
            const bytes = new Uint8Array(len);
            for (let i = 0; i < len; i++) {
                bytes[i] = binaryString.charCodeAt(i);
            }
            return bytes.buffer;
        }

        function writeString(view, offset, string) {
            for (let i = 0; i < string.length; i++) {
                view.setUint8(offset + i, string.charCodeAt(i));
            }
        }

        function toggleTheaterMode() {
            const isTheater = document.body.classList.toggle('virtual-fullscreen-active');
            const icon = document.getElementById('theaterIcon');
            if (isTheater) {
                icon.className = "fa-solid fa-compress";
            } else {
                icon.className = "fa-solid fa-expand";
            }
        }

        function closeModal(id) {
            const modal = document.getElementById(id);
            modal.classList.add('hidden');
            modal.classList.remove('flex');
        }

        function speakJapanese(text) {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel();
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'ja-JP';
                utterance.rate = 0.8; 
                window.speechSynthesis.speak(utterance);
            } else {
                showCustomAlert("Hệ thống Speech Synthesis chưa hỗ trợ trên trình duyệt của bạn!");
            }
        }
